	\\
	$a_1=3,\quad a_2=4,\quad b_1=-4,\quad b_2=-3$\\
	\\
	$a_1\,\,a_2+b_1\,\,b_2=(3)(4)+(-4)(-3)=12+12=24$\\
	\\
	$a_1\,\,a_2+b_1\,\,b_2>0$\\ 
		\\
		\begin{align*}
		\frac{a_1x+b_1y+c_1}{\sqrt{a_1^2+b_1^2}}&=\textcolor{red}{+} \frac{a_2x+b_2y+c_2}{\sqrt{a_2^2+b_2^2}}\\
		\\
		\frac{3x-4y+3}{\sqrt{(3)^2+(-4)^2}}&=\textcolor{red}{+} \frac{4x-3y+5}{\sqrt{(4)^2+(-3)^2}}\\
		\\
		\frac{3x-4y+3}{\sqrt{25}}&=\textcolor{red}{+} \frac{4x-3y+5}{\sqrt{25}}\\
		\\
		3x-4y+3&=4x-3y+5\\
		\\
		x+y+2&=0
		\end{align*}
