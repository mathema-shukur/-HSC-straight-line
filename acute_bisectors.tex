	কুমিল্লা  বোর্ড-২০২১\\ 
	$x-3y-8=0$ এবং $3x-y+7=0$ রেখাদ্বয়ের অন্তর্গত সূক্ষ্মকোণের সমদ্বিখণ্ডকের সমীকরণ নির্ণয় কর  \\
	\\
	$a_1=1,\quad a_2=3,\quad b_1=-3,\quad b_2=-1$\\
	\\
	$a_1\,\,a_2+b_1\,\,b_2=(1)(3)+(-3)(-1)=3+3=6$\\
	\\
	$a_1\,\,a_2+b_1\,\,b_2>0$\\ 
	\\
	\begin{align*}
		\frac{a_1x+b_1y+c_1}{\sqrt{a_1^2+b_1^2}}&=\textcolor{red}{-} \frac{a_2x+b_2y+c_2}{\sqrt{a_2^2+b_2^2}}\\
		\\
		\frac{x-3y-8}{\sqrt{(1)^2+(-3)^2}}&=\textcolor{red}{-} \frac{3x-y+7}{\sqrt{(3)^2+(-1)^2}}\\
		\\
		\frac{x-3y-8}{\sqrt{10}}&=\textcolor{red}{-} \frac{3x-y+7}{\sqrt{10}}\\
		\\
	x-3y-8&=-(3x-y+7)\\
		\\
		x-3y-8+3x-y+7&=0\\
		\\
		4x-4y+7&=0
	\end{align*}
